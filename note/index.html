    二分探索¶

    from bisect import bisect_left, bisect_right

# x と等しい値の探索(index)
i = bisect_left(A, x)
if i != len(A) and A[i] == x:
  return i

# x 以上の値が何個あるか(find_ge)
n = len(A) - bisect_left(A, x)

    最大公約数と最小公倍数¶

    # 最大公約数, python 3.5 から math.gcd
from fractions import gcd
gcd(a, b)

# 最小公倍数
def lcm(a, b):
  return a * b // gcd(a, b)

    ceil¶

    # math.ceil(x / y) と同じ
-(-x // y)

def ceil(x, y):
    if x % y == 0:
        return x // y
    else:
        return x // y + 1
